void setup() {
  Serial.begin(9600);
  lc.shutdown(0,false);
  lc.setIntensity(0,8);
  lc.clearDisplay(0); // Clear Displays
  lc.shutdown(1,false);
  lc.setIntensity(1,8);
  lc.clearDisplay(1); // Clear Displays
  lc.shutdown(2,false);
  lc.setIntensity(2,8);
  lc.clearDisplay(2); // Clear Displays
  lc.shutdown(3,false);
  lc.setIntensity(3,8);
  lc.clearDisplay(3); // Clear Displays
 
}
void loop() {
   while(Serial.available()>0)
  {
    String A = Serial.readString();
    if (A == "1")
    {
      a();
      }
      else if(A =="2")
      {
        b();
      }
      else if( A=="3")

      }
  }
}

void a()
{
